import { callClaudeWithImage } from './claude';

export async function extractTextFromDocument(fileBuffer: Buffer, mimeType: string): Promise<string> {
  const base64 = fileBuffer.toString('base64');
  const prompt = `Извлечи целия текст от този документ. Върни само текстът, без коментари. Ако документът е фактура, включи всички числови стойности, дати, имена.`;
  const system = 'Ти си OCR и анализатор на документи. Извличаш текст максимално точно.';
  const result = await callClaudeWithImage(prompt, base64, mimeType, system);
  return result;
}

🔧 Настройки за PSD2 (Nordigen)
Регистрирай се в Nordigen (безплатен план за до 100 сметки).

Създай API credentials – вземи SECRET_ID и SECRET_KEY.

Добави ги в .env.local:

env
NORDIGEN_SECRET_ID=your_id
NORDIGEN_SECRET_KEY=your_key
В production трябва да имаш публичен URL за webhook-а – добави в Nordigen dashboard https://your-domain.com/api/bank/webhook.
